# AgentBox egress allowlist
# POSIX basic regular expressions (Tinyproxy Filter format).
# Only domains matching these patterns can be reached from sandbox containers.
#
# Anchored patterns: ^ anchors to start, $ anchors to end, \. matches literal dot.
# [^.]+ matches one or more non-dot characters (single subdomain level).
#
# LLM API endpoints
^api\.deepseek\.com$
^[^.]+\.deepseek\.com$
^api\.anthropic\.com$
^[^.]+\.anthropic\.com$
^api\.openai\.com$
^[^.]+\.openai\.com$

# Observability (Logfire/OTel)
^api\.logfire\.dev$
^[^.]+\.logfire\.dev$

# Package repos (for sandbox tool use, if needed)
^pypi\.org$
^[^.]+\.pypi\.org$
^files\.pythonhosted\.org$
^github\.com$
^[^.]+\.github\.com$
